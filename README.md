<ul>
  <li>
    <strong>AngularJS v1</strong>
  </li>
   <li>
    <strong>NodeJS</strong>
  </li>
   <li>
    <strong>MongoDB</strong>
  </li>
</ul>
<br>

<ul>
  <li>

  </li>

</ul>
